═══ STEP 1: Create src/hooks/useIndicatorState.js ═══

This hook is similar to useAppState but parametric on the indicator data:

```js
import { useState, useMemo } from 'react'
import { getIndicatorData } from '../data/indicators-data.js'

const CE_COUNTRIES = [
  { code: 'AT', name: 'Austria' },
  { code: 'CZ', name: 'Czechia' },
  { code: 'DE', name: 'Germany' },
  { code: 'HR', name: 'Croatia' },
  { code: 'HU', name: 'Hungary' },
  { code: 'IT', name: 'Italy' },
  { code: 'PL', name: 'Poland' },
  { code: 'SI', name: 'Slovenia' },
  { code: 'SK', name: 'Slovakia' },
]

export function useIndicatorState(sheetCode) {
  const raw = getIndicatorData(sheetCode)
  const regions = raw?.regions || []
  const years = raw?.years || []

  const allCountryCodes = CE_COUNTRIES.map(c => c.code)
  const [selectedYear, setSelectedYear] = useState(years[years.length - 1] || '2024')
  const [activeCountries, setActiveCountries] = useState(allCountryCodes)
  const [focusRegions, setFocusRegions] = useState([])

  const filtered = useMemo(
    () => regions.filter(r => activeCountries.includes(r.country)),
    [regions, activeCountries]
  )

  const stats = useMemo(() => {
    const vals = filtered.map(r => r.series[selectedYear]).filter(v => v != null)
    if (!vals.length) return null
    const avg = vals.reduce((a, b) => a + b, 0) / vals.length
    const max = Math.max(...vals)
    const min = Math.min(...vals)
    return {
      avg, max, min, count: vals.length,
      maxRegion: filtered.find(r => r.series[selectedYear] === max),
      minRegion: filtered.find(r => r.series[selectedYear] === min),
    }
  }, [filtered, selectedYear])

  const focusRegionObjects = useMemo(
    () => focusRegions.map(id => regions.find(r => r.code === id)).filter(Boolean),
    [focusRegions, regions]
  )

  const focusRegionId = focusRegions[0] || null
  const setFocusRegionId = (id) => {
    if (!id) setFocusRegions([])
    else setFocusRegions(prev =>
      prev.includes(id) ? prev.filter(r => r !== id) : [...prev, id]
    )
  }

  // Compute a simple average across all filtered regions per year as "programme area avg"
  const programmeAvg = useMemo(() => {
    const result = {}
    years.forEach(y => {
      const vals = filtered.map(r => r.series[y]).filter(v => v != null)
      result[y] = vals.length ? vals.reduce((a, b) => a + b, 0) / vals.length : null
    })
    return result
  }, [filtered, years])

  return {
    // Data
    regions, years, filtered,
    // State
    selectedYear, setSelectedYear,
    activeCountries, setActiveCountries,
    focusRegions, setFocusRegions,
    focusRegionId, setFocusRegionId,
    focusRegionObjects,
    // Computed
    stats, programmeAvg,
    countries: CE_COUNTRIES,
    hasTimeSeries: years.length > 2,
  }
}
```

export default function IndicatorPage() {
  const { themeId, subId } = useParams()
  const theme = getThemeById(themeId)
  const sub = theme ? getSubThemeById(themeId, subId) : null

  // Sub-indicator selector (for Employment, Education, Unemployment, Housing)
  const [activeSheetIndex, setActiveSheetIndex] = useState(0)
  const activeSheet = sub?.dataSheets?.[activeSheetIndex] || sub?.dataSheets?.[0]

  // Hook with indicator data
  const state = useIndicatorState(activeSheet)

  if (!theme || !sub) {
    return /* error with Link back */
  }

  const meta = sub.metadata
  const hasSubIndicators = sub.subIndicators?.length > 0

  return (
    <div className={styles.page}>
      {/* Back link */}
      <Link to={`/theme/${themeId}`} className={styles.backLink}>
        ← Back to {theme.name}
      </Link>

      {/* Page header */}
      <div className={styles.header}>
        <h1>{sub.name}</h1>
        {/* type badge if needed */}
      </div>

      {/* Sub-indicator tabs (only if multiple sheets) */}
      {hasSubIndicators && (
        <div className={styles.subTabs}>
          {sub.subIndicators.map((si, idx) => (
            <button key={si.sheetCode}
              className={`${styles.subTab} ${idx === activeSheetIndex ? styles.subTabActive : ''}`}
              onClick={() => setActiveSheetIndex(idx)}
              disabled={!si.hasData}>
              {si.label}
            </button>
          ))}
        </div>
      )}

      {/* SECTION 1 — Data Source */}
      {/* SECTION 2 — Policy Context */}
      {/* SECTION 3 — Filters */}
      {/* SECTION 4 — KPI Strip */}
      {/* SECTION 5 — Map + Trend */}
      {/* SECTION 6 — Ranking */}
      {/* SECTION 7 — Data Table */}
    </div>
  )
}
